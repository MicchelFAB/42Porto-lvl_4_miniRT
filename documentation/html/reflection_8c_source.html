99double ft_schlick(t_comps *comps)
100{
101 double cos;
102 double r0;
103 double n;
104 double sin2_t;
105
106 cos = ft_dot(comps->eyev, comps->normalv);
107 if (comps->n1 > comps->n2)
108 {
109 n = comps->n1 / comps->n2;
110 sin2_t = n * n * (1 - (cos * cos));
111 if (sin2_t > 1)
112 return (1.0);
113 cos = sqrt(1.0 - sin2_t);
114 }
115 r0 = pow(((comps->n1 - comps->n2) / (comps->n1 + comps->n2)), 2);
116 return (r0 + (1 - r0) * pow((1 - cos), 5));
