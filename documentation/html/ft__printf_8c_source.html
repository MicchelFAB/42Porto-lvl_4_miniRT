15int ft_vprintf(const char fmt, va_list args)
16{
17 unsigned int prnt;
18
19 prnt = 0;
20 if (fmt == 'c')
21 prnt += ft_putcharpf(va_arg(args, int));
22 else if (fmt == 's')
23 prnt += ft_putstrpf(va_arg(args, char *));
24 else if (fmt == 'x' || fmt == 'X')
25 prnt += ft_puthexpf(va_arg(args, unsigned int), fmt);
26 else if (fmt == 'd' || fmt == 'i')
27 prnt += ft_putnbrpf(va_arg(args, int));
28 else if (fmt == 'u')
29 prnt += ft_putunbrpf(va_arg(args, unsigned int));
30 else if (fmt == 'p')
31 prnt += ft_putptrpf(va_arg(args, unsigned long long));
32 return (prnt);
33}

35int ft_printf(const char *str, ...)
36{
37 va_list args;
38 unsigned int i;
39 unsigned int result;
40
41 if (!str)
42 return (0);
43 i = 0;
44 result = 0;
45 va_start(args, str);
46 while (str[i] != '\0')
47 {
48 if (str[i] == '%')
49 {
50 if (str[i + 1] == '%')
51 result += write(1, &str[i], 1);
52 else
53 result += ft_vprintf(str[i + 1], args);
54 i++;
55 }
56 else
57 result += write(1, &str[i], 1);
58 i++;
59 }
60 va_end(args);
61 return (result);
62}
